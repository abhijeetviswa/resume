% Custom commands
\newcommand{\resumeItem}[2]{
  \item\small{
    \ifx#1\empty
        #2
    \else
        \textbf{#1}{: #2 \vspace{-2pt}}
    \fi
  }
}

\newcommand{\resumeSubheading}[5]{
  \vspace{-1pt}\item
    \begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \ifx#5\empty
        % -- Additional field empty --
        \textbf{#1} & #2 \\
        \textnormal{\small#3} & \small#4 \\
      \else
        % -- Additional field not empty --
        \textbf{#1} & #2 \\
        \textnormal{\small#3} & \small#4 \\
        \textnormal{\small#5} & \empty \\
      \fi
    \end{tabular*}\vspace{-5pt}
    
}

}
